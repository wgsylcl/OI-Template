### Dinic
```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <queue>
using namespace std;
#define MAXN 5000
#define maxm 50000
#define inf 0x7fffffff
struct edge
{
    int u, v, w, cost;
} E[(maxm + 10) << 1];
int maxflow, mincost, head[MAXN + 10], nex[(maxm + 10) << 1], dis[MAXN + 10], s, t, m, n, edge_cnt = 1, cur[MAXN + 10];
bool book[MAXN + 10], inque[MAXN + 10];
void _add(int u, int v, int w, int cost)
{
    edge_cnt++;
    E[edge_cnt] = (edge){u, v, w, cost};
    nex[edge_cnt] = head[u];
    head[u] = edge_cnt;
    return;
}
void add(int u, int v, int w, int cost)
{
    _add(u, v, w, cost);
    _add(v, u, 0, -cost);
    return;
}
bool spfa()
{
    memset(book, 0, sizeof(book));
    memset(dis, 0x3f, sizeof(dis));
    memset(inque, 0, sizeof(inque));
    queue<int> que;
    book[s] = true;
    inque[s] = true;
    que.push(s);
    dis[s] = 0;
    while (!que.empty())
    {
        int u = que.front();
        que.pop();
        for (int i = cur[u] = head[u]; i; i = nex[i])
        {
            int v = E[i].v, w = E[i].w, cost = E[i].cost;
            if (w)
            {
                if (!book[v])
                    book[v] = true;
                if (dis[v] > dis[u] + cost)
                {
                    dis[v] = dis[u] + cost;
                    if (!inque[v])
                        que.push(v), inque[v] = true;
                }
            }
        }
        inque[u] = false;
    }
    return book[t];
}
int dfs(int u, int flow)
{
    if (u == t || flow == 0)
        return flow;
    book[u] = true;
    for (int &i = cur[u]; i; i = nex[i])
    {
        int v = E[i].v, w = E[i].w, cost = E[i].cost;
        if (w && dis[u] + cost == dis[v] && !book[v])
        {
            int ret = dfs(v, min(w, flow));
            if (ret)
            {
                mincost += (cost * ret);
                E[i].w -= ret;
                E[i ^ 1].w += ret;
                book[u] = false;
                return ret;
            }
        }
    }
    book[u] = false;
    return 0;
}
void dinic()
{
    while (spfa())
    {
        memset(book, 0, sizeof(book));
        int rflow = dfs(s, inf);
        while (rflow)
        {
            maxflow += rflow;
            rflow = dfs(s, inf);
        }
    }
    return;
}
inline int read()
{
    int x = 0, f = 1;
    char ch;
    for (ch = getchar(); !isdigit(ch); ch = getchar())
        if (ch == '-')
            f = -1;
    for (; isdigit(ch); ch = getchar())
    {
        x = (x << 3) + (x << 1) + (ch - '0');
    }
    return f * x;
}
int main()
{
    n = read(), m = read(), s = read(), t = read();
    while (m--)
    {
        int u = read(), v = read(), w = read(), cost = read();
        add(u, v, w, cost);
    }
    dinic();
    printf("%d %d", maxflow, mincost);
    return 0;
}
```
