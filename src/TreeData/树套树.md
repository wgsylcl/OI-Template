# 树套树
### 线段树套平衡树 (P3380 二逼平衡树)
```cpp
#include <bits/stdc++.h>
using namespace std;
namespace IO
{
    inline int read()
    {
        int x = 0;
        char ch = getchar();
        while (ch > '9' || ch < '0')
            ch = getchar();
        while (ch <= '9' && ch >= '0')
            x = (x << 3) + (x << 1) + (ch - '0'), ch = getchar();
        return x;
    }
    void write(int x)
    {
        if (x < 0)
            putchar('-'), x = -x;
        if (x > 9)
            write(x / 10);
        putchar(x % 10 + '0');
    }
    inline void writeln(int x)
    {
        write(x);
        putchar('\n');
    }
}
using namespace IO;
#define mid ((l + r) >> 1)
const int inf = 0x7fffffff, MAXN = 5e4 + 10, MAXV = 5e7 + 10;
inline int lc(int x) { return x << 1; }
inline int rc(int x) { return x << 1 | 1; }
mt19937 rnd(time(0));
int val[MAXV], ls[MAXV], rs[MAXV], n, m, siz[MAXV], rt[MAXN << 2], idx, rd[MAXV], arr[MAXN], b[MAXN << 2], lenb;
struct node
{
    int op, l, r, k, pos;
} oper[MAXN];
namespace fhq
{
    inline int newnode(int v)
    {
        rd[++idx] = rnd();
        siz[idx] = 1;
        val[idx] = v;
        return idx;
    }
    inline void update(int x) { siz[x] = siz[ls[x]] + siz[rs[x]] + 1; }
    void split(int u, int v, int &x, int &y)
    {
        if (!u)
        {
            x = y = 0;
            return;
        }
        if (v < val[u])
        {
            y = u;
            split(ls[u], v, x, ls[u]);
        }
        else
        {
            x = u;
            split(rs[u], v, rs[u], y);
        }
        update(u);
    }
    int merge(int x, int y)
    {
        if (!x || !y)
            return x + y;
        if (rd[x] <= rd[y])
        {
            rs[x] = merge(rs[x], y);
            update(x);
            return x;
        }
        else
        {
            ls[y] = merge(x, ls[y]);
            update(y);
            return y;
        }
    }
    inline int qrnk(int x, int k)
    {
        while (siz[ls[x]] + 1 != k)
            if (k <= siz[ls[x]])
                x = ls[x];
            else
                k -= siz[ls[x]] + 1, x = rs[x];
        return x;
    }
    inline void insert(int &root, int v)
    {
        int x, y;
        split(root, v - 1, x, y);
        root = merge(merge(x, newnode(v)), y);
    }
    inline void del(int &root, int v)
    {
        int x, y, z;
        split(root, v - 1, x, y);
        split(y, v, y, z);
        root = merge(merge(x, merge(ls[y], rs[y])), z);
    }
    inline int getfrnk(int &root, int v)
    {
        int x, y;
        split(root, v - 1, x, y);
        int res = siz[x];
        root = merge(x, y);
        return res;
    }
    inline int prev(int &root, int v)
    {
        int x, y;
        split(root, v - 1, x, y);
        int res = siz[x] ? val[qrnk(x, siz[x])] : -inf;
        root = merge(x, y);
        return res;
    }
    inline int more(int &root, int v)
    {
        int x, y;
        split(root, v, x, y);
        int res = siz[y] ? val[qrnk(y, 1)] : inf;
        root = merge(x, y);
        return res;
    }
}
void build(int x, int l, int r)
{
    for (int i = l; i <= r; i++)
        fhq::insert(rt[x], arr[i]);
    if (l >= r)
        return;
    build(lc(x), l, mid);
    build(rc(x), mid + 1, r);
}
void modify(int x, int l, int r, int p, int v)
{
    fhq::del(rt[x], arr[p]);
    fhq::insert(rt[x], v);
    if (l == p && p == r)
    {
        arr[p] = v;
        return;
    }
    if (p <= mid)
        modify(lc(x), l, mid, p, v);
    else
        modify(rc(x), mid + 1, r, p, v);
}
int getfrnk(int x, int l, int r, int pl, int pr, int v)
{
    if (r < pl || l > pr)
        return 0;
    if (pl <= l && r <= pr)
        return fhq::getfrnk(rt[x], v);
    return getfrnk(lc(x), l, mid, pl, pr, v) + getfrnk(rc(x), mid + 1, r, pl, pr, v);
}
inline int qrnk(int k, int pl, int pr)
{
    int l = 1, r = lenb, ans;
    while (l <= r)
    {
        if (getfrnk(1, 1, n, pl, pr, b[mid]) + 1 <= k)
            ans = b[mid], l = mid + 1;
        else
            r = mid - 1;
    }
    return ans;
}
int prev(int x, int l, int r, int pl, int pr, int v)
{
    if (r < pl || l > pr)
        return -inf;
    if (pl <= l && r <= pr)
        return fhq::prev(rt[x], v);
    return max(prev(lc(x), l, mid, pl, pr, v), prev(rc(x), mid + 1, r, pl, pr, v));
}
int more(int x, int l, int r, int pl, int pr, int v)
{
    if (r < pl || l > pr)
        return inf;
    if (pl <= l && r <= pr)
        return fhq::more(rt[x], v);
    return min(more(lc(x), l, mid, pl, pr, v), more(rc(x), mid + 1, r, pl, pr, v));
}
int main()
{
    n = read(), m = read();
    for (int i = 1; i <= n; i++)
        b[++lenb] = arr[i] = read();
    build(1, 1, n);
    int opt, l, r, k, pos;
    for (int i = 1; i <= m;i++)
        switch (oper[i].op = read())
        {
        case 1:
            oper[i].l = read(), oper[i].r = read(), oper[i].k = read();
            break;
        case 2:
            oper[i].l = read(), oper[i].r = read(), oper[i].k = read();
            break;
        case 3:
            oper[i].pos = read(), oper[i].k = read();
            b[++lenb] = oper[i].k;
            break;
        case 4:
            oper[i].l = read(), oper[i].r = read(), oper[i].k = read();
            break;
        case 5:
            oper[i].l = read(), oper[i].r = read(), oper[i].k = read();
            break;
        }
    sort(b + 1, b + 1 + lenb);
    lenb = unique(b + 1, b + 1 + lenb) - b - 1;
    for (int i = 1; i <= m;i++)
        switch (oper[i].op)
        {
        case 1:
            writeln(getfrnk(1, 1, n, oper[i].l, oper[i].r, oper[i].k) + 1);
            break;
        case 2:
            writeln(qrnk(oper[i].k, oper[i].l, oper[i].r));
            break;
        case 3:
            modify(1, 1, n, oper[i].pos, oper[i].k);
            break;
        case 4:
            writeln(prev(1, 1, n, oper[i].l, oper[i].r, oper[i].k));
            break;
        case 5:
            writeln(more(1, 1, n, oper[i].l, oper[i].r, oper[i].k));
            break;
        }
        return 0;
}
```
