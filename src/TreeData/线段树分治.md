# 线段树分治

```admonish question title="[P5787 二分图 /【模板】线段树分治](https://www.luogu.com.cn/problem/P5787)"
神犇有一个 $n$ 个节点的图。

因为神犇是神犇，所以在 $k$ 时间内有 $m$ 条边会出现后消失。

神犇要求出每一时间段内这个图是否是二分图。

$n,k = 10^5$，$m = 2\times 10^5$。$1 \le x,y \le n$，$0 \le l \le r \le k$。
```

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef pair<int, int> pii;
const int MAXN = 1e5 + 10, MAXV = MAXN << 2, MAXM = 2e5 + 10;
inline int ls(int x) { return x << 1; }
inline int rs(int x) { return x << 1 | 1; }
struct edge
{
    int u, v;
} e[MAXM];
// fy -> fx
struct node
{
    int fx, fy;
    bool added;
};
vector<int> tr[MAXV];
stack<node> stk;
int f[MAXN * 2], h[MAXN * 2], n, m, k;
inline int getf(int x)
{
    while (x != f[x])
        x = f[x];
    return x;
}
inline void merge(int x, int y)
{
    x = getf(x), y = getf(y);
    if (h[x] < h[y])
        swap(x, y);
    f[y] = f[x];
    stk.push({x, y, h[y] == h[x]});
    if (h[y] == h[x])
        h[x]++;
}
inline void del()
{
    auto [fx, fy, added] = stk.top();
    stk.pop();
    h[fx] -= added;
    f[fy] = fy;
}
void update(int x, int l, int r, int pl, int pr, int v)
{
    if (pl <= l && r <= pr)
    {
        tr[x].push_back(v);
        return;
    }
    int mid = (l + r) >> 1;
    if (pl <= mid)
        update(ls(x), l, mid, pl, pr, v);
    if (mid < pr)
        update(rs(x), mid + 1, r, pl, pr, v);
    return;
}
void solve(int x, int l, int r)
{
    for (int i = 0; i < tr[x].size(); i++)
    {
        int u = e[tr[x][i]].u, v = e[tr[x][i]].v;
        int fu = getf(u), fv = getf(v);
        if (fu == fv)
        {
            for (int j = i; j; j--)
                del(), del();
            for (int i = l; i <= r; i++)
                cout << "No\n";
            return;
        }
        merge(u, v + n);
        merge(v, u + n);
    }
    if (l >= r)
        cout << "Yes\n";
    else
    {
        int mid = (l + r) >> 1;
        solve(ls(x), l, mid);
        solve(rs(x), mid + 1, r);
    }
    for (int i = tr[x].size(); i; i--)
        del(), del();
}
int main()
{
    ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
    cin >> n >> m >> k;
    for (int i = 1; i <= n * 2; i++)
        f[i] = i, h[i] = 1;
    for (int l, r, i = 1; i <= m; i++)
    {
        cin >> e[i].u >> e[i].v >> l >> r;
        update(1, 1, k, l + 1, r, i);
    }
    solve(1, 1, k);
    return 0;
}
```
